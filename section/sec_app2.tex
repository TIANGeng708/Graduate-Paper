\noindent \textbf{引理~3.4（带标签噪声的Cross Entropy）}：
\begin{multline}

\end{proof}

\noindent \textbf{引理~3.5（标签噪声下Cross Entropy的随机梯度）}：
\begin{multline}\label{eq:lamma5}
F_n\left(\boldsymbol{\omega};\zeta_n^{noi}\right)=\boldsymbol{g}_n\left(\boldsymbol{\omega},\zeta_n^{cle}\right)+\frac{1}{|\zeta_n^{noi}|}\sum_{\left(x_j,\hat{y}_j\right)\in \zeta_n^{noi}}r_j\left(\frac{\nabla_{\boldsymbol{\omega}} M\left(x_j, y_j; \boldsymbol{\omega}\right)}{M\left(x_j, y_j; \boldsymbol{\omega}\right)} -\frac{\nabla_{\boldsymbol{\omega}} M\left(x_j, \hat{y}_j; \boldsymbol{\omega}\right)}{M\left(x_j, \hat{y}_j; \boldsymbol{\omega}\right)}\right)
\end{multline}

\begin{proof}
\begin{equation}
    \begin{aligned}
&g_n\left(\boldsymbol{\omega};\zeta_n^{noi}\right)\\
&=\nabla_\omega F_n\left(\boldsymbol{\omega};\zeta_n^{noi}\right)\\
&=\nabla_\omega\left(F_n\left(\boldsymbol{\omega};\zeta_n^{cle}\right)+\frac{1}{|\zeta_n^{noi}|}\sum_{\left(x_j,\hat{y}_j\right)\in \zeta_n^{noi}}r_j\left(-log\left(\frac{M\left(x_j, \hat{y}_j; \boldsymbol{\omega}\right)}{M\left(x_j, y_j; \boldsymbol{\omega}\right)}\right)\right)\right)\\
&=\nabla_\omega\left(F_n\left(\boldsymbol{\omega};\zeta_n^{cle}\right)\right)+\nabla_\omega\left(\frac{1}{|\zeta_n^{noi}|}\sum_{\left(x_j,\hat{y}_j\right)\in \zeta_n^{noi}}r_j\left(-log\left(\frac{M\left(x_j, \hat{y}_j; \boldsymbol{\omega}\right)}{M\left(x_j, y_j; \boldsymbol{\omega}\right)}\right)\right)\right)\\
&=\boldsymbol{g}_n\left(\boldsymbol{\omega},\zeta_n^{cle}\right)+\frac{1}{|\zeta_n^{noi}|}\sum_{\left(x_j,\hat{y}_j\right)\in \zeta_n^{noi}}r_j\left(\frac{\nabla_{\boldsymbol{\omega}} M\left(x_j, y_j; \boldsymbol{\omega}\right)}{M\left(x_j, y_j; \boldsymbol{\omega}\right)} -\frac{\nabla_{\boldsymbol{\omega}} M\left(x_j, \hat{y}_j; \boldsymbol{\omega}\right)}{M\left(x_j, \hat{y}_j; \boldsymbol{\omega}\right)}\right)
    \end{aligned}
\end{equation}
\end{proof}

\noindent \textbf{命题~1.4（Cross-Entorpy损失下含标签噪声的随机梯度的平方的期望）}
：在假设 \ref{ass:gradient} 下，任意客户端 $n\in\N$ 在噪声数据集 $\Dnoi_n$ 上的 $\gb_n\left(\omegab,\zetanoi_n\right)$ 是有界的，即：
\begin{equation}
    \mathbb{E}_{\zetanoi_n\sim \Dnoi_n}\left[\left\Vert \gb_n\left(\omegab, \zetanoi_n\right) \right\Vert^2\right] \leq G_n^2 + H_4^2,
    \tag{xxx}
\end{equation}
其中 $H_4^2$ 定义为：
\begin{align}\label{eq:H4}
    H_4^2 = 4\delta\frac{\Delta}{Z}+4\delta^2\left(1-\frac{\Delta}{Z}\right)
    \tag{xxx}
\end{align}

\begin{proof}
为简化推导，我们将
$$
\mathbb{E}_{\zetanoi_n\sim \Dnoi_n}[\Vert \gb_n(\omegab, \zetanoi_n) \Vert^2]
$$
表示为 $\mathbb{E}_r\mathbb{E}_{\zetacle_n}[\Vert \gb_n(\omegab, \zetanoi_n) \Vert^2]$，其中 $r \sim \text{Bernoulli}(\delta)$ 表示随机噪声。
